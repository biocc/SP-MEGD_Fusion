#轻链
count_lc=0
for(i in 1:length(peptides))
{
  print(i)
  nAA=nchar(peptides[i])
  j=1
  k=0
  while(j<=(nAA-6))
  {
    mer=substr(peptides[i],j,j+6)
    if(grepl(mer,LC))
    {
      j=j+1
      k=k+1
      count_lc=count_lc+1
      loc=str_locate(LC,mer)
      if(k==1)
      {
        lc[1,loc[1]:loc[2]]=lc[1,loc[1]:loc[2]]+1
      }
      else
      {
        lc[1,loc[2]]=lc[1,loc[2]]+1
      }
    }
    else
    {
      if(k!=0)
      {
        j=j+7
      }
      else
      {
        j=j+1
      }
      k=0
    }
  }
}

    }
  }
}
